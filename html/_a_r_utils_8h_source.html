59  // depending on whether or not we want pixels or points, run the correct function.
60  // Note that, when points are requested, they are for some reason, the opposite of what they should be.

62  screenBounds = [[UIScreen mainScreen] nativeBounds];
63  }else{
64  screenBounds = [[UIScreen mainScreen] bounds];
65  }
66

68  float width,height;
69 
70  // function to set width and height - takes the odd behavior associated with requesting points
71  // into account.

73 
74  if(!useNative){
75  width = screenBounds.size.height;
76  height = screenBounds.size.width;
77  }else{
78  width = screenBounds.size.width;
79  height = screenBounds.size.height;
80  }
81 
82  };

84  // Set the dimensions as appropriate depending on our orientation.
85  // Note that for some reason, and I'm not sure if it's an oF, IOS or mistake on my part, but the first time
86  // it enters this switch block, dimensions are off, so there is an nested if statement to try and fix that in
87  // some of the cases.
88  switch(UIDevice.currentDevice.orientation){
89  case UIDeviceOrientationFaceUp:
90  setWidthAndHeight();
91 
92  // if face up - we just assume portrait
93  dimensions.x = width;
94  dimensions.y = height;
95  break;
96 
97  case UIDeviceOrientationFaceDown:
98  setWidthAndHeight();
99 
100  // if face up - we just assume portrait
101  dimensions.x = width;
102  dimensions.y = height;
103  break;
104  case UIInterfaceOrientationUnknown:
105  // if unknown - we just assume portrait
106  dimensions.x = width;
107  dimensions.y = height;
108  break;
109 
110  // upside down registers, but for some reason nothing happens and there might be weirdness :/
111  // leaving this here anyways but probably best to just disable upsidedown portrait.
112  case UIInterfaceOrientationPortraitUpsideDown:
113  setWidthAndHeight();
114

116  dimensions.y = height;
